113 void GASolver::cross_over(bool always) {
114  if (population.size() < 2){
115  return;
116  }
117  float cross_over_occur = ceil(population.size() * cross_over_rate);
118  vector<int> list_fertile = {};
119  for (int i = 0; i < population.size(); ++i) {
120  list_fertile.push_back(i);
121  }
122  for (int i = 0; i < cross_over_occur; i++) {
123  int rnd = rand()%list_fertile.size();
124  int a = list_fertile.at(rnd);
125  rnd = rand()%list_fertile.size();
126  int b = list_fertile.at(rnd);
127  for (int index = 0; index < list_fertile.size(); index++) {
128  if (list_fertile.at(index) == b || list_fertile.at(index) == a){
129  list_fertile.erase(list_fertile.begin() + index);
130  }
131  }
132  vector <GAChromosome> offspring = GAChromosome::cross_over(population.at(a), population.at(b));
133  GAChromosome offspring_a = offspring.at(0);
134  GAChromosome offspring_b = offspring.at(1);
135  if ((offspring_a.error < min(population.at(a).error, population.at(b).error)
136  && offspring_b.error < min(population.at(a).error, population.at(b).error))
137  || always){
138  population.at(a) = offspring_a;
139  population.at(b) = offspring_b;
140  }
141  }
142 }
